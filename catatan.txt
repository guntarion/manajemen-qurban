CREATE TABLE progres_sapi (
    id INT AUTO_INCREMENT,
    nomer INT NOT NULL,
    urutan INT,
    lingkar INT,
    panjang INT,
    bobot INT,
    tipe VARCHAR(15),
    sembelih BOOLEAN DEFAULT FALSE,
    note VARCHAR(200),
    PRIMARY KEY (id)
);

CREATE TABLE pengqurban (
    ID INT AUTO_INCREMENT,
    kode_kirim VARCHAR(200),
    nama_pengqurban VARCHAR(200),
    nama_peruntukan VARCHAR(200),
    no_hape INT,
    no_qurban INT,
    no_proses INT,
    alamat_jalan VARCHAR(250),
    alamat_rt VARCHAR(10),
    alamat_rw VARCHAR(10),
    pesan_khusus VARCHAR(500),
    keterangan VARCHAR(500),
    is_sembelih BOOLEAN DEFAULT FALSE,
    time_sembelih DATETIME,
    is_ready BOOLEAN DEFAULT FALSE,
    time_ready DATETIME,
    is_kirim BOOLEAN DEFAULT FALSE,
    time_kirim DATETIME,
    is_done BOOLEAN DEFAULT FALSE,
    time_done DATETIME,
    tipe_1 VARCHAR(30),
    tipe_2 VARCHAR(30),
    tipe_3 VARCHAR(30),
    PRIMARY KEY (ID)
);

CREATE TABLE lembaga (
    ID INT AUTO_INCREMENT,
    kode_kirim VARCHAR(200),
    nama_elemen VARCHAR(200),
    nama_pic_1 VARCHAR(200),
    nama_pic_2 VARCHAR(200),
    no_hape_pic_1 INT,
    no_hape_pic_2 INT,
    alamat_jalan VARCHAR(250),
    alamat_rt VARCHAR(10),
    alamat_rw VARCHAR(10),
    keterangan_1 VARCHAR(500),
    keterangan_2 VARCHAR(500),
    tanda_terima VARCHAR(1000),
    isDiterima BOOLEAN DEFAULT TRUE,
    request_ori INT,
    prosen_modif INT,
    riil_delivery INT,
    is_ready BOOLEAN DEFAULT FALSE,
    time_ready DATETIME,
    is_kirim BOOLEAN DEFAULT FALSE,
    time_kirim DATETIME,
    is_done BOOLEAN DEFAULT FALSE,
    time_done DATETIME,
    tipe_1 VARCHAR(30),
    tipe_2 VARCHAR(30),
    tipe_3 VARCHAR(30),
    PRIMARY KEY (ID)
);
